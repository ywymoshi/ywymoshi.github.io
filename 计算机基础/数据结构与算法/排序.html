# 1. 冒泡排序

原理：从第一个数据开始依次比较相邻元素的大小，如果前者大于后者，则将大的元素往后移动， 每次将最大的数据放在最后，多次迭代完成排序.

性能: 空间复杂度 O(1)
最好时间复杂度 O(n) ps：当数组刚好是顺序的。
最坏时间复杂度 O(n^2) ps：当数组顺序逆序。
数据平均时间复杂度 O(n^2) ps : 当数据杂乱无章时

代码：

function bubblingSort (arr = []) {
  let len = arr.length; 
  let temp;
  for (let i = 0; i < len; i++) {
    for (let j = 1; j < len - i; j++) {
      if (arr[j] > arr[j + 1]) {
         temp = arr[j + 1];
         arr[j + 1 ] = arr[j];
         arr[j] = temp;
      }     
    }   
  }
  return arr; 
}

# 2. 选择排序

原理： 从未排序的序列中每次查找到最小的元素，然后再替换。

性能: 空间复杂度 O(1)
最坏 、最好、平均时间复杂度 O(n^2) ps：当数组任意顺序。

    function selectionSort (arr = []) {
      let len = arr.length; 
      let minIndex;
      let temp;
      for (let i = 0; i < len; i++) {
        minIndex = i;
        for (let j = i + 1; j < len; j++) {  
          if (arr[j] < arr[minIndex]) {
            minIndex = j;
          }     
        }
        temp = arr[minIndex];
        arr[minIndex] = arr[i];
        arr[i] = temp;
      }
      return arr; 
    }

# 3. 插入排序

原理：选取未排序元素，插入到已排序区间的合适位置，直到未排序元素为空，从左至右维护一个已排序好的数组。 将第一个元素定义为已排序。

性能: 空间复杂度 O(1)
最好时间复杂度 O(n) ps：当数组刚好是顺序的。
最坏时间复杂度 O(n^2) ps：当数组顺序逆序。
数据平均时间复杂度 O(n^2) ps : 当数据杂乱无章时

     function insertSort (arr = []) {
       let len = arr.length; 
       let temp;
       for (let i = 1; i < len; ++ i) {
         temp = arr[i];
         let j = i - 1;
         for (; j >= 0; -- j) {
           if (arr[j] < temp) {
              arr[j + 1] = arr[j];
            } else {
              break;            
           }
         }   
          arr[j + 1] = temp;
       }
       return arr; 
     }

# 4.快速排序

原理：选取一个元素为基点，指定两个指针0 与 length - 1,将左边比基点大的与右边比基点小的进行交换。 将第一个元素定义为已排序。

性能: 空间复杂度 O(1)
最好时间复杂度 O(nlogn) ps：如果每次选取分区点时，都能选中中位数，把数组等分成两个，那么此时的时间复杂度和归并一样，都是 O(n*logn)。
最坏时间复杂度 O(n^2) ps：也就是如果每次分区都选中了最小值或最大值，得到不均等的两组。 数据平均时间复杂度 O(nlogn) ps : 当数据杂乱无章时

function quickSort(arr = []) {
  const sort = (arr, low, high) => {
    let i = low, j = high;
    if (low >= high) return;
    let temp = arr[low];
    while (i != j) {
      while (arr[j] >= temp && j > i) --j;
      arr[i] = arr[j];
      while (arr[i] <= temp && j > i) ++i;
      arr[j] = arr[i];
    }
    arr[i] = temp;
    sort(arr, i + 1, high);
    sort(arr, low, i - 1);
  }
  sort(arr, 0, arr.length - 1);
  return arr;
}

# 5. 归并排序

原理：它首先将数组不断地二分，直到最后每个部分只包含 1 个数据。然后再对每个部分分别进行排序，最后将排序好的相邻的两部分合并在一起，这样整个数组就有序了。 将第一个元素定义为已排序。

性能: 空间复杂度 O(n)
最好、最坏、平均时间复杂度 O(nlogn) ps：每次的迭代，需要对两个有序数组进行合并，这样的动作在 O(n) 的时间复杂度下就可以完成。因此，**归并排序的复杂度就是二者的乘积 O(nlogn)。**同时，它的执行频次与输入序列无关，因此，归并排序最好、最坏、平均时间复杂度都是 O(nlogn)。

function mergeSort(arr = []) {
  const sort = (R) => {
    if (R.length < 2 ) return R;
    let mid = Math.floor(R.length / 2);
    let left = R.slice(0, mid);
    let right = R.slice(mid);
    return merge(sort(left), sort(right));
  }
  const merge = (left, right) => {
    let result = [];
    while (left.length && right.length) {
      if (left[0] < right[0])
        result.push(left.shift());
      else
        result.push(right.shift());
    }
    while (left.length) {
      result.push(left.shift());
    }
    while (right.length) {
      result.push(right.shift());
    }
    return result;
  }
  const result = sort(arr);
  return result;
}
