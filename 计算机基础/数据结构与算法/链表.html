# js实现双向链表

    function Node(element) {
        this.element =element;
        this.next = null;
        this.prev =null;
    }

    class LinkedList {
        constructor(props) {
            this.head = null;
            this.length = 0;
        }
        getElementAt(position) {
            if (position < 0 || position >= this.length)
              return null;
            let current = this.head;
            for (let i = 0; i < position; i++) 
              current = current.next;
            return current;
        }
        append (element) {
            let node = new Node(element);
            if (this.head == null) {
                this.head = node;
            } else {
              let current = getElementAt(this.length - 1);
                  node.prev = current;
                  current.next = node;
            }
            this.length ++; 
        } 
        insert(position, element) {
            if (position < 0 || position >= this.length)
              return false;
            let node = new Node(element);
            if (position === 0) {
                node.next = this.head;
                this.head = node;
            } else {
              let current = getElementAt(position - 1);
                  node.prev = current;
                  node.next = current.next;
                  current.next = node;
            }
            this.length ++; 
        }
        removeAt (position) {
            if (position < 0 || position >= this.length)
              return false;
            let current = this.head;
            if (position === 0) {
                this.head = current.next;
            } else {
              let previous = getElementAt(position - 1);
                  current = previous.next;
                  current.next.prev = previous;
                  previous.next = next.next;
            }
            this.length --; 
        }
        indexOf (element) {
          let current = this.head;
          for (let i = 0; i < this.length;i++) {
              if (current.element = element)
                return i;
              current = current.next;
          }
        }
        isEmpty (){
          return this.length === 0;
        }
        size () {
          return this.length;
        }
        getHead () {
          return this.head;
        }
        clear () {
          this.head = null;
          this.length = 0;
        }
        toString () {
        let current = this.head;
        let s = '';
        while (current) {
            let next = current.next;
            let prev = current.prev;
            next = next ? next.element : 'null';
            prev = prev ? prev.element : 'null';
            s += `element: ${current.element},prev:${prev},next: ${next};`;
            current = current.next;
        }
        return s;
        }
    }
